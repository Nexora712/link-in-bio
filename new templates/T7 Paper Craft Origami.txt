<script>
document.querySelectorAll('.paper-link').forEach(link => {
  link.addEventListener('click', function() {
    const url = this.dataset.url;
    if (url) {
      // Paper fold animation
      this.style.transform = 'rotate(0deg) translateY(-5px) scale(0.98)';
      setTimeout(() => {
        this.style.transform = 'rotate(0deg) translateY(-5px) scale(1)';
        window.open(url, '_blank');
      }, 200);
    }
  });
});
</script>
